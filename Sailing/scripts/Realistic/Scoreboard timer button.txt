integer DISPLAY_STRING      = 204000; integer DISPLAY_EXTENDED    = 204001; integer REMAP_INDICES       = 204002; integer RESET_INDICES       = 204003; integer SET_FADE_OPTIONS    = 204004; integer SET_FONT_TEXTURE    = 204005; integer SET_LINE_COLOR      = 204006; integer SET_COLOR           = 204007; integer RESCAN_LINKSET      = 204008;integer CurrentTime =0;integer Duration=100;///60*60*24 - 1 dayinteger Enable_Timer = TRUE;//Scoreboard timer button//Written by brian Young/ Daijo zenovka//goes in scoreboard timer buttonstring Delimiter = "%$%^";integer Sailing_Scoreboard_Realistic = -307; integer Sailing_Scoreboard_Wacky = -308; string Time_Pass = "Time_Pass";default{    state_entry()    {        llMessageLinked(LINK_THIS,DISPLAY_STRING,"Sailboard Timer","1");    }    touch_start(integer total_number)    {        if (Enable_Timer)        {            llOwnerSay("timer begin");            llMessageLinked(LINK_THIS,DISPLAY_STRING,"Enabled","0");            llSetTimerEvent(1.0);        }        else        {            llOwnerSay("timer stop");            llMessageLinked(LINK_THIS,DISPLAY_STRING,"Disabled","0");            CurrentTime = 0;            llSetTimerEvent(0);           }        Enable_Timer=!Enable_Timer;    }    timer()    {        CurrentTime++;                if (CurrentTime >= Duration)        {            llOwnerSay("timer set now");            CurrentTime = 0;            llRegionSay(Sailing_Scoreboard_Realistic, Time_Pass +  Delimiter);            llRegionSay(Sailing_Scoreboard_Wacky, Time_Pass +  Delimiter);        }    }}
