//windindicator Laser(Realistic) Script//By Brian Young/Daijo Zenovka//Belongs to primitive 'windindicator' of child of Boat root//with references to the Flying Tako Sailboat Script//It is used to receive information about wind direction, and display it as a particle stream emitting from the windindicator primitive. It is located in windindicator primitive.vector windDir;list particleVars=[PSYS_PART_FLAGS,            PSYS_PART_FOLLOW_SRC_MASK,              PSYS_SRC_PATTERN,            PSYS_SRC_PATTERN_DROP,            PSYS_PART_MAX_AGE,3.0,            PSYS_PART_START_COLOR,<1,1,1>,            PSYS_PART_START_ALPHA,1.0,            PSYS_PART_START_SCALE,<.3,.3,.3>,            PSYS_SRC_BURST_RATE,0.1,            PSYS_SRC_BURST_SPEED_MIN,0.5,            PSYS_SRC_BURST_SPEED_MAX,0.5,            PSYS_SRC_BURST_PART_COUNT,1];list tmpVars;integer AVATAR_NAME = 5;            //set the particle stream based on the direction of the Second Life WindsetParticles(){    windDir=llWind(<0,0,0>);    tmpVars=particleVars;    tmpVars+=[PSYS_SRC_ACCEL, windDir];    llParticleSystem(tmpVars);}   default {    state_entry()     {        llParticleSystem([]);    }    timer()     {        setParticles();    }        link_message(integer sender_num, integer num, string str, key id)     {               if (num==1000)         {//Reset particle system            llParticleSystem([]);            llSetTimerEvent(0);            llSetText("",<0,0,0>,0);        }        else if (num>1000)         {                       }           else {            if (num == AVATAR_NAME)            {//set the name of the boat at top of mast                llSetText(str,<1,0,0>,1);            }            else            {//set the timer call count                llSetTimerEvent((float)num);            }        }    }}
