//Sax sound Script //By Brian Young/Daijo Zenovka//Goes in root prim of Saxophoneinteger Sax_Channel = -310;string Delimiter = "%$%^";string Play_Sound = "Play_Sound";string CurrentSound;//Songsstring Race_Theme = "race theme";string Race_Wacky_Theme = "race wacky theme";string Sailing_End = "sailing end";string Sailing_Start = "sailing start";string Sailing_Theme = "sailing theme";string Sailing_Wacky_Theme = "sailing wacky theme";PlaySound(string TheSound){    llStopSound();     llTriggerSound(TheSound, 1.0);}LoopSound(string TheSound){    llStopSound();    llLoopSoundMaster(TheSound, 1.0);}default{    state_entry()    {        llStopSound();        llListen(Sax_Channel,"", NULL_KEY, "");    }    listen( integer channel, string name, key id, string message )    {        if (channel == Sax_Channel)        {            list temp = llParseString2List(message, [Delimiter], []);                                    if (llList2String(temp,0) == Play_Sound)            {                string NewSound = llList2String(temp,1);                                if (llList2Integer(temp,2) == TRUE)                {                    LoopSound(NewSound);                                    }                else                {                    PlaySound(NewSound);                                    }                CurrentSound = NewSound;            }                    }            }}
